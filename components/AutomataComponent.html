import { Component, OnInit } from '@angular/core';
import * as xml2js from 'xml2js';
/**
 * Para hacer el parseo de xml a html se necesita AJAX con el directorio /assets, que es donde tenemos el .xml origen. Por eso se debe importar HttpClient
 */
import { HttpClient, HttpHeaders } from '@angular/common/http';
@Component({
  selector: 'app-automata',
  templateUrl: './automata.component.html',
  styleUrls: ['./automata.component.css'],
})
export class AutomataComponent implements OnInit {
  /**
   * Bloques para hacer el parseo de xml desde el directorio /assets al template.
   */
  public xmlItems: any;
  public xmlItemsUnidades: any;
  public xmlItemsEquipamientos: any;

  constructor(private _http: HttpClient) {}

  loadXML(event: any) {
    this._http
      /*Indicar origen del archivo*/
      .get('/assets/automatas-ejemplo.xml', {
        headers: new HttpHeaders()
          .set('Content-Type', 'text/xml')
          .append('Access-Control-Allow-Methods', 'GET')
          .append('Access-Control-Allow-Origin', '*')
          .append(
            'Access-Control-Allow-Headers',
            'Access-Control-Allow-Headers, Access-Control-Allow-Origin, Access-Control-Request-Method'
          ),
        responseType: 'text',
      })
      .subscribe((data) => {
        /*Paso a valor el contenido del 'value' del template*/
        let valor = event; //.target.value;
        this.parseXML(data, valor).then((data) => {
          this.xmlItems = data;
        });
      });
  }
  parseXML(data: any, event: any) {
    return new Promise((resolve) => {
      var k: string | number,
        arr: any = [],
        parser = new xml2js.Parser({
          trim: true,
          explicitArray: true,
        });
      parser.parseString(data, function (_err, result) {
        var obj = result.automats;
        for (k in obj.automat) {
          var item = obj.automat[k];

          /*Bucle que recorre el array con los valores enviados diferenciadores para los automatas*/
          for (let i in event) {
            if (item.$.name.includes(event[i])) {
              arr.push({
                automata: item.$.name,
              });
            }
          }
        }
        resolve(arr);
      });
    });
  }

  loadXMLUnidades(event: any) {
    this._http
      /*Indicar origen del archivo*/
      .get('/assets/automatas-ejemplo.xml', {
        headers: new HttpHeaders()
          .set('Content-Type', 'text/xml')
          .append('Access-Control-Allow-Methods', 'GET')
          .append('Access-Control-Allow-Origin', '*')
          .append(
            'Access-Control-Allow-Headers',
            'Access-Control-Allow-Headers, Access-Control-Allow-Origin, Access-Control-Request-Method'
          ),
        responseType: 'text',
      })
      .subscribe((data) => {
        let valor = event;
        this.parseXMLUnidades(data, valor).then((data) => {
          this.xmlItemsUnidades = data;
        });
      });
  }

  parseXMLUnidades(data: any, event: any) {
    return new Promise((resolve) => {
      var k: string | number;
      var j: string | number,
        arrUnidades: any = [],
        parser = new xml2js.Parser({
          trim: true,
          explicitArray: true,
        });
      parser.parseString(data, function (_err, result) {
        var obj = result.automats;
        for (k in obj.automat) {
          var item = obj.automat[k];
          //console.log('evento=', event);
          //console.log('item.$.name=', item.$.name);
          if (item.$.name.includes(event.target.innerHTML)) {
            for (j in item.unit) {
              let unidad = item.unit[j];
              //console.log('esto:', unidad.$['min-occurs']);
              arrUnidades.push({
                unidad: unidad.$.type,
                numero: unidad.$['min-occurs'],
              });
            }
          }
        }
        resolve(arrUnidades);
      });
    });
  }

  loadXMLEquipamientos(event: any) {
    this._http
      /*Indicar origen del archivo*/
      .get('/assets/Units-ejemplo.xml', {
        headers: new HttpHeaders()
          .set('Content-Type', 'text/xml')
          .append('Access-Control-Allow-Methods', 'GET')
          .append('Access-Control-Allow-Origin', '*')
          .append(
            'Access-Control-Allow-Headers',
            'Access-Control-Allow-Headers, Access-Control-Allow-Origin, Access-Control-Request-Method'
          ),
        responseType: 'text',
      })
      .subscribe((data) => {
        /*Paso a valor el contenido del 'value' del template*/
        let valor = event;
        this.parseXMLEquipamientos(data, valor).then((data) => {
          this.xmlItemsEquipamientos = data;
        });
      });
  }

  parseXMLEquipamientos(data: any, event: any) {
    return new Promise((resolve) => {
      var k: string | number;
      var j: string | number,
        arrEquipamientos: any = [],
        parser = new xml2js.Parser({
          trim: true,
          explicitArray: true,
        });
      parser.parseString(data, function (_err, result) {
        var obj = result.units;
        //console.log('item.$.name=', result.units);
        for (k in obj.unit) {
          let numeroRecorte = 5;
          let objetivo = event.target.innerHTML.slice(5);
          var item = obj.unit[k];
          //console.log('event=', event);
          //console.log(item);
          //console.log('item.$.name=', item.$.name);
          //console.log('event.target.innerHTML=', event.target.innerHTML);
          //console.log('objetivo=', objetivo);
          if (objetivo.includes(item.$.name)) {
            for (j in item.equipments) {
              let equipos = item.equipments[j];
              for (let i in equipos.equipment) {
                let unidad = equipos.equipment[i];
                //console.log('esto:', unidad);
                arrEquipamientos.push({
                  unidad: unidad.$.type,
                  //numero: unidad.$['min-occurs'],
                });
              }
            }
          }
        }
        resolve(arrEquipamientos);
      });
    });
  }

  ngOnInit(): void {}
}
